52/// Private marker trait for erasable generics - types with this trait have the same
53/// repr for all generic param values, and can therefore be transmuted on
54/// the singular Repr type representation on ABI boundaries.
55///
56/// # Safety
57/// This type must only be implemented on types known to be repr equivalent
58/// to their Repr type.
59// #[cfg_attr(
60//     wbg_diagnostic,
61//     diagnostic::on_unimplemented(
62//         label = "generic parameter is not a valid Wasm Bindgen ErasableGeneric type",
63//         note = "\nRecommendation: Add the direct `: wasm_bindgen::JsGeneric` convenience trait bound for JsValue generics, instead of `ErasableGeneric`.\n",
64//     )
65// )]
66pub unsafe trait ErasableGeneric {
67    /// The singular concrete type that all generic variants can be transmuted on
68    type Repr: 'static;
69}
70
71unsafe impl<T: ErasableGeneric> ErasableGeneric for &mut T {
72    type Repr = &'static mut T::Repr;
73}
74
75unsafe impl<T: ErasableGeneric> ErasableGeneric for &T {
76    type Repr = &'static T::Repr;
77}

79/// Trait bound marker for types that are passed as an own generic type.
80/// Encapsulating the ErasableGeneric invariant that must be maintained, that
81/// the repr of the type is the type of the concrete target type repr.
82/// This is useful to provide simple debuggable trait bounds for codegen.
83#[cfg_attr(
84    wbg_diagnostic,
85    diagnostic::on_unimplemented(
86        message = "Unable to call function, since the concrete generic argument or return value cannot be type-erased into the expected generic repr type for the function",
87        label = "passed concrete generic type does not match the expected generic repr type",
88        note = "Make sure that all erasable generic parameters satisfy the trait bound `ErasableGeneric` with the correct repr. Wasm Bindgen generic parameters and return values for functions are defined to work for specific type-erasable generic repr types only.",
89    )
90)]
91pub trait ErasableGenericOwn<ConcreteTarget>: ErasableGeneric {}

93impl<T, ConcreteTarget> ErasableGenericOwn<ConcreteTarget> for T
